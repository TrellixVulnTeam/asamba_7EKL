414  """
415  As the name explains, this function receives a record array of GYRE modes summary, and trims/clips
416  the modes based on their period spacing pattern
417 
418  @param modes: The observed modes, where each mode in the list is an instance of the "star.mode" class
419  @type modes: list of star.mode
420  @param rec_gyre: The numpy record array from GYRE frequency list coming from one GYRE output file
421  @type rec_gyre: np.recarray
422  @param dic_mode_types: Look up dictionary to match the modes identification (l, m) with the modes.id_type

425  @type dic_mode_types: dict
426  @param trim_delta_freq_factor: This is the fraction of the frequency difference (delta_f) between the

432  @return: False if, for one among many reasons, it is not possible to trim the GYRE list based on the
433  observed modes. If it succeeds, the input GYRE list will be trimmed to match the size of the input
434  modes, and then it will be returned.
435  @rtype: np.recarray or bool
436  """
437  _trim_modes_by_dP(modes, rec_gyre, dic_mode_types, trim_delta_freq_factor)

496 def _split_learning_sets(self):
497  """
498  Refer to the documentation of the public method split_learning_set().
499  """
500  p_train = float(self.training_percentage)
501  p_cv = float(self.cross_valid_percentage)
502  p_test = float(self.test_percentage)
503  percentages = [p_train, p_cv, p_test]
504  if any(p < -1e-15 for p in percentages):
505  logger.error('_split_learning_sets: Change the default (-1) for self.*_percentage=-1')
506  sys.exit(1)
507 
508  if not all(0 <= p <= 1 for p in percentages):
509  logger.error('_split_learning_sets: All three self.*_percentage must be set between 0 and 100')
510  sys.exit(1)
511 
512  if np.abs(1-(p_train + p_cv + p_test)) > 1e-5:
513  logger.error('_split_learning_sets: The three self.*_percentage do not all up to 1.0 (+/- 1e-5)')
514  sys.exit(1)
515 
516  n_learn = self.sample_size
517  n_train = int(n_learn * p_train)
518  n_cv = int(n_learn * p_cv)
519  n_test = int(n_learn * p_test)
520 
521  # due to round-off, the sum of splitted sets may not add up to sample_size, then ...
522  if n_learn != n_train + n_cv + n_test:
523  n_train = n_learn - (n_cv + n_test)
524  self.set('training_size', n_train)
525  self.set('cross_valid_size', n_cv)
526  self.set('test_size', n_test)
527 
528  # Make randomly shuffled indixes for slicing
529  ind_learn = np.arange(n_learn)
530  np.random.shuffle(ind_learn)
531 
532  # reset various arrays according to the new indixes
533  self.set('ids_models', self.ids_models[ind_learn])
534  self.set('ids_rot', self.ids_rot[ind_learn])
535 
536  ind_train = ind_learn[ : n_train]
537  ind_cv = ind_learn[n_train : n_train + n_cv]
538  ind_test = ind_learn[n_train + n_cv :]
539 
540  # slice the learning set into training/cross-validation/test sets
541  learn_x = np.empty_like(self.learning_x)
542  learn_y = np.empty_like(self.learning_y)
543  learn_x[:] = self.learning_x
544  learn_y[:] = self.learning_y
545 
546  self.set('training_x', learn_x[ind_train])
547  self.set('training_y', learn_y[ind_train])
548 
549  self.set('cross_valid_x', learn_x[ind_cv])
550  self.set('cross_valid_y', learn_y[ind_cv])
551 
552  self.set('test_x', learn_x[ind_test])
553  self.set('test_y', learn_y[ind_test])
554 
555  # In some casas, the log_Teff and log_g values for the learning set is available too, which we fix now
556  self.set('training_log_Teff', self.learning_log_Teff[ind_train])
557  self.set('cross_valid_log_Teff', self.learning_log_Teff[ind_cv])
558  self.set('test_log_Teff', self.learning_log_Teff[ind_test])
559 
560  self.set('training_log_g', self.learning_log_g[ind_train])
561  self.set('cross_valid_log_g', self.learning_log_g[ind_cv])
562  self.set('test_log_g', self.learning_log_g[ind_test])
563 
564  logger.info('_split_learning_sets" Training, Cross-Validation and Test sets are prepared')
