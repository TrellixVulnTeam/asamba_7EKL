33  development is to avoid creating/closing a connection/cursor to the database everytime one
34  freaking model ID needs be fetched. This avoids connection overheads when thousands to
35  millions of track IDs need be retrieved.

38  db_def.grid_db(dbname) object.

145 def find_missing_models(dbname, eta):
146  """
147  For a given rotation rate, eta, find the models.id that do not have a corresponding row in the
148  modes.id_model list. These are models which the GYRE computation has either failed, or for some
149  reason, they are not still inserted into the "modes" table yet.
150 
151  @param dbname: The name of the database, e.g. "grid"
152  @type dbname: string
153  @return: list of models.id where the GYRE computaiton shall be either repeated, or the data must
154  be inserted into the modes table.
155  @rtype: list of int
156  """
157  tup_rot = (eta, )
158  dic_rot = get_dic_look_up_rotation_rates_id(dbname)
159  try:
160  id_rot = dic_rot[tup_rot]
161  tup_id_rot = (id_rot, )
162  logger.info('find_missing_models: corresponding id_rot is "{0}"'.format(id_rot))
163  except:
164  logger.error('find_missing_models: eta={0} is invalid, and not supported yet!'.format(eta))
165  sys.exit(1)
166 
167  with db_def.grid_db(dbname=dbname) as the_db:
168  cmnd = 'select id from models'
169  the_db.execute_one(cmnd, None)
170  id_from_models = [tup[0] for tup in the_db.fetch_all()]
171  n = len(id_from_models)
172 
173  cmnd = 'select distinct on (id_model) id_model from modes where id_rot=%s group by id_model'
174  the_db.execute_one(cmnd, tup_id_rot)
175  id_from_modes = [tup[0] for tup in the_db.fetch_all()]
176  m = len(id_from_modes)
177 
178  # Sanity checks
179  if n == 0:
180  logger.error('find_missing_models: The "models" table is empty!')
181  sys.exit(1)
182  if m == 0:
183  logger.error('find_missing_models: The "modes" table is empty!')
184  sys.exit(1)
185  if m > n:
186  logger.error('find_missing_models: The funny case that we have more input to modes than the input models!')
187  sys.exit(1)
188 
189  # The safe mode
190  if n == m:
191  logger.info('find_missing_models: All Input models have a corresponding mode list for eta="{0}"'.format(eta))
192  return None
193  else:
194  missing = set(id_from_models).symmetric_difference(set(id_from_modes))
195  n_missing = len(missing)
196  logger.info('find_missing_models: Returning missing {0} models.id values'.format(n_missing))
197  return sorted(list(missing))

200 def find_missing_gyre_task(dbname, eta, h5_prefix='ad-sum'):
201  """
202  For a given rotation rate, eta, find the names of the GYRE input files, and the missing GYRE output
203  files for which the GYRE output file is absent in the database (so, may need to do the GYRE
204  computaitons for these).
205 
206  @param dbname: The name of the database to connect to
207  @type dbname: string
208  @param eta: the rotation rate in percentage
209  @type eta: float
210  @param h5_prefix: The prefix which is added at the begining of the HDF5 GYRE output files to
211  distinguish the adiabatic/non-adiabatic summary/mode files. Set to '' to ignore it
212  @type h5_prefix: str
213  @return: two lists of strings are returned.
214  1. The first list is the core name of the GYRE input files, e.g.
215  M35.000-ov0.010-Z0.010-logD00.00-MS-Xc0.6092-00983.gyre
216  2. The second list is the core name of the GYRE ouput files, e.g.
217  ad-sum-M35.000-ov0.010-Z0.010-logD00.00-MS-Xc0.6092-00983-eta50.00.h5
218  @rtype: tuple of two lists
219  """
220  missing = find_missing_models(dbname=dbname, eta=eta)
221  # tups = [(val, ) for val in missing]
222  cmnd = 'select M_ini, fov, Z, logD, model_number, Xc from tracks inner join models on tracks.id = models.id_track where models.id=%s'
223  result = []
224  with db_def.grid_db(dbname=dbname) as the_db:
225  for i, id_model in enumerate(missing):
226  tup = (id_model, )
227  the_db.execute_one(cmnd, tup)
228  result.append(the_db.fetch_one())
229 
230  list_gyre_in = []
231  list_gyre_out = []
232  for i, tup in enumerate(result):
233  M_ini, fov, Z, logD, model_number, Xc = tup
234  core = 'M{0:6.3f}-ov{1:4.3f}-Z{2:4.3f}-logD{3:05.2f}-MS-Xc{4:5.4f}-{5:05d}'.format(
235  M_ini, fov, Z, logD, Xc, model_number)
236  gyre_in = 'M{0:06.3f}/gyre_in/{1}.gyre'.format(M_ini, core)
237  gyre_out = 'M{0:06.3f}/gyre_out/eta{1:05.2f}/{2}-{3}-eta{4:05.2f}.h5'.format(
238  M_ini, eta, h5_prefix, core, eta)
239 
240  list_gyre_in.append(gyre_in)
241  list_gyre_out.append(gyre_out)
242 
243  return list_gyre_in, list_gyre_out

485 def get_track_id(dbname_or_dbobj, M_ini, fov, Z, logD):
486  """

488 
489  @param dbname_or_dbobj: The first argument of this function can have two possible types. The reason

491  type of the input argument, which we benefit from here. The reason behind this choice of
492  development is to avoid creating/closing a connection/cursor to the database everytime one
493  freaking track ID needs be fetched. This gives a nice speedup when thousands to millions of
494  track IDs need be retrieved.
495  The two possible inputs are:
496  - dbname: string which specifies the name of the dataase. This is used to instantiate the

503  @type fov: float
504  @param Z: initial metallicity
505  @type Z: float
506  @param logD: the logarithm of the diffusive mixing coefficient
507  @type logD: float
508  @return: the id of the corresponding row, if the row exists, and if the query succeeds.
509  In case of a failure, we return False
510  @rtype: integer
511  """
512  cmnd = 'select id from tracks where M_ini~%s and fov~%s and Z~%s and logD~%s'
513  tup = (M_ini, fov, Z, logD)
514 
515  if isinstance(dbname_or_dbobj, str):
516  with db_def.grid_db(dbname=dbname_or_dbobj) as the_db:
517  the_db.execute_one(cmnd, tup)
518  result = the_db.fetch_one()
519  #
520  elif isinstance(dbname_or_dbobj, db_def.grid_db):

543 def get_tables_info(dbname):
544  """
545  Retrieve the information of the tables in the database passed by its name (as dbname). The following
546  informations are retrieved, and used as the key of the returned dictionary:
547  - user_name
548  - schema_name
549  - table_name
550  - index_name
551  - is_unique
552  - is_primary
553  - index_type
554  - indkey
555  - index_keys
556  - is_functional
557  - is_partial
558 
559  Note that the value corresponding to each key is a list of strings, and the length of all these returned
560  lists are identical.
561 
562  @param dbname: the database name
563  @type dbname: string
564  @return: a dictionary with the entire information, accessed through 11 keys listed above. The associated
565  value of each key is a list of strings
566  @rtype: dict
567  """
568  cmnd = 'SELECT \
569  U.usename AS user_name, \
570  ns.nspname AS schema_name, \
571  idx.indrelid :: REGCLASS AS table_name, \
572  i.relname AS index_name,\
573  idx.indisunique AS is_unique, \
574  idx.indisprimary AS is_primary, \
575  am.amname AS index_type, \
576  idx.indkey, \
577  ARRAY( \
578  SELECT pg_get_indexdef(idx.indexrelid, k + 1, TRUE) \
579  FROM \
580  generate_subscripts(idx.indkey, 1) AS k \
581  ORDER BY k \
582  ) AS index_keys, \
583  (idx.indexprs IS NOT NULL) OR (idx.indkey::int[] @> array[0]) AS is_functional, \
584  idx.indpred IS NOT NULL AS is_partial \
585  FROM pg_index AS idx \
586  JOIN pg_class AS i \
587  ON i.oid = idx.indexrelid \
588  JOIN pg_am AS am \
589  ON i.relam = am.oid \
590  JOIN pg_namespace AS NS ON i.relnamespace = NS.OID \
591  JOIN pg_user AS U ON i.relowner = U.usesysid \
592  WHERE NOT nspname LIKE %s; -- Excluding system tables'
593  val = ('pg%', )
594 
595  with db_def.grid_db(dbname=dbname) as the_db:
596  the_db.execute_one(cmnd, val)
597  result = the_db.fetch_all()
598 
599  if result is None:
600  logger.error('get_tables_info: failed')
601  sys.exit(1)
602  n = len(result)
