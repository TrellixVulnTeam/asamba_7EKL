23  """
24  Constructor that stores the mass, metalicity, overshoot and extra diffusive mixing per each
25  track. By default, the attributes are set to "-1.0" as a physically meaningless initial value,
26  to facilitate more convenient capturing of wrong initializations. E.g. you can initialize a track
27  as:
28 
29  >>>a_track = var_def.track(M_ini=12.0, Z=0.014, fov=0.024, logD=2.25)
30 
31  @param M_ini: the initial mass in solar unit
32  @type M_ini: float
33  @param Z: metallicity (where the solar metallicity from Asplund et al. 2009 is 0.014)
34  @type Z: float
35  @param fov: the exponential overshoot free parameter (see e.g. Eq. 2 in Moravveji et al.
36  2016, ApJ)
37  @type fov: float
38  @param logD: the (logarithm of the) constant diffusive mixing in the radiative envelope.
39  See e.g. Fig. 2a in Moravveji et al. (2016, ApJ)
40  @type logD: float
41  """
42  self.M_ini = M_ini
43  self.Z = Z
44  self.fov = fov
45  self.logD = logD

201  def set_track_parameters(self):
202  """
203  Glob and find all available tracks that are organized inside the repository (hence dir_repos).
204  The tracks are organized based on their initial mass, and lie inside the "hist" subdirectory, e.g.
205  "dir_repos/M12.345/hist/M12.345-ov0.010-Z0.018-logD01.23.hist"
206  Then, the whole track parameters will be stored into the "tracks" object
207 
208  @param self: an instance of the var_def.tracks() object
209  @type self: class object
210  """
211  if self.n_dirs_M_ini == 0:
212  logger.error('set_track_parameters: first call set_mass_directories()')
213  sys.exit(1)
214 
215  list_dirs_M_ini = self.get_list_dirs_M_ini()
216 
217  list_track_paths = []
218 
219  hist_search_pattern = self.get_hist_search_pattern()
220  if hist_search_pattern == '':
221  logger.error('set_track_parameters: first call set_hist_search_pattern()')
222  sys.exit(1)
223 
224  # Collect all available hist files
225  for dr in list_dirs_M_ini:
226  hist_search = dr + hist_search_pattern
227  hists = glob.glob(hist_search)
228  n_hists = len(hists)
229  if n_hists == 0:
230  logger.error('var_def: no history files found in the path: "{0}"'.format(hist_search))
231  sys.exit(1)
232  list_track_paths += hists[:]
233 
234  # Extract parameters from history file paths
235  # Store info the class objects
236  list_tracks = []
237  n_tracks = len(list_track_paths)
238  hist_extension = self.hist_extension
239 
240  for i, trck in enumerate(list_track_paths):
241  ind_slash = trck.rfind('/')
242  ind_point = trck.rfind(hist_extension)
243  corename = trck[ind_slash+1 : ind_point]
244  params = corename.split('-')
245  n_params = len(params)
246  if n_params != 4:
247  logger.error('var_def: the number of retrieved parameters is different from expected')
248  sys.exit(1)
249 
250  M_ini = float(params[0][1:])
251  fov = float(params[1][2:])
252  Z = float(params[2][1:])
253  logD = float(params[3][4:])
254 
255  one_track = track(M_ini=M_ini, Z=Z, fov=fov, logD=logD)
256  one_track.set_filename(trck)
257  list_tracks.append(one_track)
258 
259  # Store the data into the "tracks" object
260  self.set_n_tracks(n_tracks)
261  self.set_list_tracks(list_tracks)
262  logger.info('set_track_parameters: Setting track numbers and list done.')
