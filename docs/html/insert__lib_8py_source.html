41 def prepare_insert_models():
42  """
43  "Prepare" a command that allows inserting any row into the models table. This is to facilitate
44  much faster interaction with the database.
45  """
46  other_attrs = var_lib.get_model_other_attrs()
47  n_other = len(other_attrs)
48 
49  # The 4 track attributes are retrieved by the id_track, so we must skip them. The other three

59 
60  cmnd = 'prepare prepare_insert_models ({0}) as \
61  insert into models ({1}) values ({2})'.format(
62  str_types, str_attrs, str_qmarks)

535 def model_line_to_model_object(line):
536  """
537  Convert one row of the ascii parameter file, into an instance of the var_def.model class, and store
538  the line values into those of the corresponding attribute. The returned object is useful to insert
539  into the SQL database
540 
541  @param line: one line of the parameter file, which contains numbers only
542  @type line: string
543  @return: an instance of the var_def.model class, with the attributes all set from the input line.
544  @rtype: object
545  """
546  if not isinstance(line, str):
547  logger.error('model_line_to_model_object: The input is not a string')
548  sys.exit(1)
549  line = line.rstrip('\r\n').split()
550 
551  # instantiate a model object
552  model = var_def.model()
553 
554  # get the model attribute names
555  avail_attrs = var_lib.get_model_attrs()
556 
557  for i, attr in enumerate(avail_attrs):
558  conv = float
559  if attr == 'model_number': conv = int
560  val = conv(line[i])
561  setattr(model, attr, val)
562 
563  return model
