149  def solve_normal_equation(self):
150  """
151  Find the analytic solution for the unknown hypothesis coefficients \f$\theta\f$, which minimizes the
152  cost function \f$ J(\theta) \f$ as defined below.
153 
154  \f[ J(\theta)= \frac{1}{2m} (\theta^T X-y)^T \cdot (\theta^T X-y) \f]
155 
156  For more information refer to:
157  <a href="http://eli.thegreenplace.net/2014/derivation-of-the-normal-equation-for-linear-regression">Click to Open</a>
158  Consequently, the analytic solution to \f$\theta\f$ is:
159 
160  \f[ \theta_0 = (X^T \cdot X)^{-1} \cdot X^{-1} \cdot y. \f]
161 
162  A brief remark on the dimensionality of the terms: For a learning set of size \f$ m\f$, with \f$ n+1\f$ features
163  (including the intercept coefficient), and for the observed/trained output \f$ y \f$ being a matrix of \f$ m\times K\f$
164  (for \f$ K\f$ modes), then the coefficient matrix \f$ \theta_0\f$ is \f$ (n+1) \times K \f$.
165 
166  Once \f$\theta_0\f$ is analytically derived, then the cost function is minimized. If we assume
167  this set of coefficients make the cost function approach zero \f$J(\theta_0)\approx 0\f$, intuitively
168  \f$ \theta_0^T\cdot X \approx y \f$.
169 
170  One can immediately solve for the unknown feature vector \f$ X \f$, which reproduces the observations \f$ y_0\f$,
171  given the corresponding coefficients \f$ \theta_0 \f$. To that end, we multiply both sides of the last equation
172  by \f$ \theta \f$, followed by a multiplication with \f$ (\theta_0 \cdot \theta_0^T)^{-1} \f$ to yield \f$ X \f$:
173 
174  \f[ X_0 \approx (\theta_0 \cdot \theta_0^T)^{-1} \cdot (\theta \cdot y_0) \f]
175 
176  Needless to highlight that \f$X_0\f$ is a vector of size \f$(n+1)\f$, for an intercept and \f n\f$ features.
177 
178  Notes:
179  - The resulting coefficients are saved as the following attribute self.normal_equation_theta, and the resulting
180  feature vector \f$ X_0 \f$ is stored as the attribute self.normal_equation_features.
181  - The model frequencies \f$ y \f$ and the observed frequencies \f$ y_0 \f$ are converted to the per day
182  (\f$ d^{-1} \f$) unit for a fair comparison.
183  - "A major drawback of the Maximum Likelihood approach is that it is vulnerable to overfitting, because no care
184  is taken for cmplex models that try to learn the specificities of the particular training set (Theodoridis, S.
185  2015, Machine Learning book)."
186 
187  @param self: instance of the neural_net class
188  @type self: object
189  """
190  _solve_normal_equation(self)
191  self.set('normal_equation_done', True)

194  def chi_square(self):
195  """
196  We define the \f$ \chi^2\f$ score between the i-th observed frequency \f$ f_i^{\rm (obs)}, and the model
197  frequency \f$ f_i^{\rm (mod)}\f$ (coming from e.g. the learning set) as the following
198 
199  \f[ \chi^2 = \sum_{i=1}^{K} \chi^2_i = \sum_{i=1}^{K} \frac{1}{2}
200  \left(\frac{f_i^{\rm (obs) - f_i^{\rm (mod)}{\sigma_i^2}\right)^2.
201  \f]
202 
203  Here, \f$ K\f$ is the total number of observed modes, and \f$\sigma_i\f$ is the 1-\f$\sigma\f$ uncertainty
204  around each observed frequency.
205 
206  @param self: An instance of the neural_net() class
207  @type self: obj
208  @return: the "MAP_chi_square_matrix" and "MAP_chi_square" attributes of the class will be set
209  @rtype: None
210  """
211  _chi_square(self)
212  self.set('chi_square_done', True)

215  def max_a_posteriori(self):
216  """
217  This routine finds the attributes of the model which maximizes the posterior likelihood function, hence MAP.
218  It consists of the following steps:
219 
220  1. Priors: Either set uniformly, or are set based on a comparison between log_Teff and log_g of the model and
221  the star. For each hypothesis \f$h\f$, we return the prior information \f$P(h)\f$, and \f$\ln P(h)\f$.
222 
223  2. LogLikelihood, or chi square \f$\chi^2\f$: the natural logarithm of the probability density of the data
224  given the hypothesis, i.e.
225  \f$\chi^2=\ln P(D|h) = \frac{1}{2K}\sum_{i=1}^{K} \left((f_i^{\rm (obs)} - f_i^{\rm (mod)})/sigma_i \right)^2 \f$.
226  A recommended option here is to "rescale" the chi-square values to avoid numerical overflow.
227 
228  3. Evidence, which is basically an inner dot product between the prior vector and the likelihood vector:
229  \f$P(D)=\sum_h P(D\h) P(h) \f$. We return both \f$P(D)\f$ and \f$\ln P(D)\f$.
230 
231  4. Posterior \f$P(h|D)\f$: Based on the Bayes Theorem, the posterior is
232 
233  \f[
234  \frac{P(h|D)}{s}=\frac{\frac{P(D|h)}{s}P(h)}{P(D)},
235  \f]
236 
237  where \f$s\f$ is an optional "scaling" factor used to "rescale" the loglikelihood. Indeed, setting \f$s=1\f$
238  recovers the Bayes theorem in its original form. This scaling is allowed, since we only make relative
239  comparison between the models.
240 
241  @param self: an instance of the neural_net() class
242  @type self: obj
243  """
244  _max_a_posteriori(self)

300 def _solve_normal_equation(self):
301  """
302  Refer to the documentation of solve_normal_equation() for further details.
303  """
304  # Get THE instance of the sampling class already stored in neural_net object
305  # sample = self.get('sampling')
306  # Check if the sampling is already done or not
307  if not self.learning_done:
308  logger.error('_solve_normal_equation: The sampling is not done yet. Call sampler.sampling.build_sampling_sets() first')
309  sys.exit(1)
310 
311  x = self.learning_x # (m, n)
312  x = utils.prepend_with_column_1(x) # (m, n+1)
313  y = self.learning_y # (m, K)
314 
315  a = np.dot(x.T, x) # (n+1, n+1)
316  b = np.linalg.inv(a) # (n+1, n+1)
317  c = np.dot(x.T, y)
318  d = np.dot(b, c)
319 
320  theta = d[:] # (n+1, K)
321  self.set('normal_equation_theta', theta)
322 
323  # observed frequency from list of modes
324  modes = self.modes
325 
326  # Now, solve for the "best-value" features using theta from above
327  freqs = np.array([ mode.freq for mode in modes ]).T # (K, 1)
328 
329  e = np.linalg.inv(np.dot(theta, theta.T)) # (n+1, n+1)
330  f = np.dot(theta, freqs) # (n+1, K) x (K, 1) == (n+1, 1)
331  g = np.dot(e, f) # (n+1, n+1) x (n+1, 1) == (n+1, 1)
332 
333  self.set('normal_equation_features', g)
334 
335  h = np.dot(g.T, theta) - freqs
336  J = old_div(np.dot(h.T, h), (2 * len(freqs)))
337 
338  self.set('normal_equation_cost', J)

374 def _set_priors(self):
375  """
376  Refer to the documentation below the max_a_posteriori() method for further details
377  """
378  bool_arr = np.array([self.MAP_uniform_prior, self.MAP_use_log_Teff_log_g_prior])
379  n_bool = np.sum(bool_arr * 1)
380  if n_bool != 1:
381  logger.error('_set_priors: set only one of "MAP_uniform_prior" or "MAP_use_log_Teff_log_g_prior" to True:')
382  sys.exit(1)
383 
384  learning_y = self.get('learning_y')
385  m, K = learning_y.shape
386 
387  if self.MAP_uniform_prior:
388  prior = old_div(np.ones(m), float(m))
389  elif self.MAP_use_log_Teff_log_g_prior:
390  # get observed log_Teff and log_g together with their errors from the star
391  if self.log_Teff_err_lower == 0 or self.log_Teff_err_upper == 0:
392  logger.error('_set_priors: set log_Teff_err_lower and log_Teff_err_upper first')
393  sys.exit(1)
394 
395  obs_log_Teff = self.log_Teff
396  obs_log_Teff_err = np.max([ self.log_Teff_err_lower, self.log_Teff_err_upper ])
397  obs_log_g = self.log_g
398  obs_log_g_err= np.max([ self.log_g_err_lower, self.log_g_err_upper ])
399  if not all([ obs_log_Teff != 0, obs_log_Teff_err != 0,
400  obs_log_g != 0, obs_log_g_err != 0 ]):
401  logger.error('_set_priors: Specify the log_Teff, log_g and their errors properly')
402  sys.exit(1)
403 
404  lrn_log_Teff = self.learning_log_Teff[:]
405  lrn_log_g = self.learning_log_g[:]
406 
407  prior_log_Teff = utils.gaussian(x=lrn_log_Teff, mu=obs_log_Teff, sigma=obs_log_Teff_err)
408  prior_log_g = utils.gaussian(x=lrn_log_g, mu=obs_log_g, sigma=obs_log_g_err)
409 
410  prior = prior_log_Teff * prior_log_g
411 
412  ln_prior = np.log(prior)
413 
414  self.set('MAP_prior', prior)
415  self.set('MAP_ln_prior', ln_prior)

418 def _chi_square(self):
419  """
420  Refer to the documentation below the chi_square() method for further details
421  """
422  modes = self.modes
423  freqs = np.array([ mode.freq for mode in modes ])
424  sigma = np.array([ mode.freq_err for mode in modes ])
425  sigma *= self.frequency_sigma_factor
426  n_freq = len(freqs)
427 
428  learning_y = self.get('learning_y') # (m, K)
429  m, K = learning_y.shape
430  try:
431  assert n_freq == K
432  except AssertionError:
433  logger.error('_chi_square: The number of observed frequencies ({0}) not equal to the number of columns of learning_y ({1})'.format(n_freq, K))
434  sys.exit(1)
435 
436  # duplicate/reapeat the frequency vector m times to form identical matrix to learning_y
437  freqs_matrix = np.tile(freqs, m).reshape(m, K) # (m, K)
438  sigma_matrix = np.tile(sigma, m).reshape(m, K) # (m, K)
439 
440  # calculate the chi^2 in a vectorized form, i.e. element-by-element operation in abstract form
441  chi_2_matrix = old_div(np.power(old_div((freqs_matrix - learning_y),sigma_matrix), 2.0), 2.0)
442  chi_2 = np.sum(chi_2_matrix, axis=1)
443 
444  # save the above two chi squares as attributes of the class
445  self.set('MAP_chi_square_matrix', chi_2_matrix)
446  self.set('MAP_chi_square', chi_2)

449 def _set_likelihood(self):
450  """
451  For the full documentation refer to the max_a_posteriori() method
452  """
453  # star = self.get('sampling').get('star')
454  modes = self.get('modes')
455  errs = np.array([ mode.freq_err for mode in modes ])
456  chi_2 = self.get('MAP_chi_square')
457  K = len(chi_2)
458  ln_L = -K/2.0 * np.log(2.0*np.pi) - np.sum(np.log(errs)) - chi_2
459 
460  self.set('MAP_ln_likelihood', ln_L)
461  self.set('MAP_likelihood', np.exp(ln_L))

464 def _rescale_ln_likelihood(self):
465  """
466  Rescale the product of the ln(likelihood) x ln(prior), to avoid numerical underflow when evaluating
467  the evidence. Note that one cannot rescale, e.g. only the ln_likelihood, else the weighting of the
468  rescaled ln(likelihood) with unscaled ln(prior) will render the problem as wrong.
469 
470  The scaling factor is in natural logarithm. The value of ln_scale is chosen so that the sum of
471  unscaled ln(likelihood) + lin(prior) adds up to zero, after rescaling. Once ln_scale is found, then
472  ln(likelihood) and ln(prior) are each rescaled by 0.5 x ln_scale. Subsequently, ln(likelihood),
473  likelihood, ln(prior) and prior are all updated.
474  """
475  if not self.rescale_ln_likelihood: return
476 
477  ln_P_D_h = self.get('MAP_ln_likelihood')
478  ln_P_h = self.get('MAP_ln_prior')
479  ln_prod = ln_P_D_h + ln_P_h
480 
481  ln_scale = np.max(ln_prod) / 2.0
482  ln_P_D_h += ln_scale
483  ln_P_h += ln_scale
484 
485  self.set('MAP_ln_likelihood', ln_P_D_h)
486  self.set('MAP_likelihood', np.exp(ln_P_D_h))
487  self.set('MAP_ln_prior', ln_P_h)
488  self.set('MAP_prior', np.exp(ln_P_h))
489 
490  logger.info('_set_likelihood: rescaling ln(likelihood) x ln(prior) with ln_scale={0:.2f}'.format(ln_scale))

493 def _set_evidence(self):
494  """
495  Refer to the documentation below the max_a_posteriori() method for further details
496  """
497  P_h = self.get('MAP_prior') # == P(h)
498  P_D_h = self.get('MAP_likelihood') # P(D|h)
499  # P_D = np.sum(P_D_h * prior) # P(D)
500  P_D = np.dot(P_D_h, P_h)
501 
502  self.set('MAP_evidence', P_D)
503  self.set('MAP_ln_evidence', np.log(P_D))

507  """
508  Refer to the documentation below the max_a_posteriori() method for further details

523 def _marginalize_wrt(self, wrt):
524  """
525  For the full documentation refer to the marginalize_wrt() method.
526  """
527  # sample = self.get('sampling')
528  x_names = self.get('feature_names')
529  if wrt not in x_names:

532

539  arr_wrt = np.array(list(set_wrt))
540  arr_wrt = np.sort(arr_wrt)
541  post_wrt= np.zeros(n_wrt) # to store the marginalized posterior probabilities
542 
543  post = self.get('MAP_posterior')
544 
545  for i_wrt, val_wrt in enumerate(arr_wrt):
546  ind_wrt = np.where(rec[wrt] == val_wrt)[0]
547  post_wrt[i_wrt] = np.sum(post[ind_wrt])
548 
549  return (arr_wrt, post_wrt)
