22 def list_to_recarray(list_input, dtype):
23  """
24  Convert a list of tuples to a numpy recordarray. Each tuple is one retrieved row of data from calling
25  the SQL queries, and fetching them through e.g. db.fetch_all() method.
26 
27  @param list_input: The inputs to be converted to numpy record array.
28  @type list_input: list
29  @return: recarray with the number of rows equal to the number of tuples in the input list, and the
30  number of columns equal to the number of items in each tuple. The dtype for each column is
31  passed as an input argument by the user.
32  @rtype: np.recarray
33  """
34  if not is_py3x:
35  names = [tup[0] for tup in dtype]
36  types = [tup[1] for tup in dtype]
37  dtype = [(name.encode('ascii'), dtp) for name, dtp in zip(names, types)]
38 
39  try:
40  a = np.core.records.fromarrays(np.array(list_input).T, dtype=dtype)
41  # a = np.core.records.fromarrays(list_to_ndarray(list_input), dtype=dtype)
42  except Exception as xpt:
43  print('error occured:', xpt)
44  else:
45  return a

107 def gaussian(x, mu, sigma):
108  """
109  Return the Normal (Gaussian) probability distribution function g(x) for x with mean mu and standard
110  deviation sigma, following the definition
111 
112  \f[
113  N(x,\mu,\sigma)=\frac{1}{\sqrt{2\pi}\sigma}\exp\left[-\frac{1}{2}\left(\frac{x-\mu}{\sigma}\right)^2\right].
114  \f]
115 
116  @param x: array or value of the input
117  @type x: ndarray or float
118  @param mu: the mean of the population
119  @type mu: float
120  @param sigma: the standard deviation of the population around the mean
121  @type sigma: npdarray or float
122  @return: The probability of x being between the interval x and x+epsilon, where epsilon goes to zero
123  @rtype: ndarray or float
124  """
125  if isinstance(x, np.ndarray) and isinstance(sigma, np.ndarray):
126  try:
127  assert len(x) == len(sigma)
128  except AssertionError:
129  logger.error('gaussian: the size of input arrays "x" and "sigma" must be identical')
130  sys.exit(1)
131 
132  scale = old_div(1.,(np.sqrt(2*np.pi)*sigma))
133  arg = -0.5 * (old_div((x - mu)**2,sigma**2))
134 
135  return scale * np.exp(arg)
