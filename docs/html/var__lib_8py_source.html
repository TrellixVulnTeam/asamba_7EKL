227 def get_model_parameters_from_gyre_out_filename(filename):
228  """
229  Extract all parameters in the MESA output/GYRE input file, and return them as a tuple. A random file
230  may look like the following:
231  /home/user/my_grid/M12.345/gyre_out/eta25.00/ad-sum-M12.345-ov0.012-Z0.014-logD02.50-MS-Xc0.3217-00312-eta25.00.h5
232 
233  @param filename: The full path to the GYRE output file
234  @type filename: string
235  @return: a tuple with 7 parameters of the model as float or integer values. The order of the
236  output is the following:
237  - 0: M_ini, initial mass
238  - 1: fov, exponential overshoot parameter
239  - 2: Z, initial metallicity
240  - 3: logD: the logarithm of the extra diffusive mixing
241  - 4: Xc: the core hydrogen mass fraction
242  - 5: model_number: integer, giving MESA step number
243  - 6: eta: rotation rate in percentage w.r.t. the break up rotation rate
244  @rtype: tuple
245  """
246  ind_slash = filename.rfind('/')
247  ind_point = filename.rfind('.')
248  corename = filename[ind_slash+1 : ind_point].split('-')
249 
250  n_prefix = 2
251  pos_M = n_prefix
252  pos_fov = pos_M + 1
253  pos_Z = pos_fov + 1
254  pos_logD = pos_Z + 1
255  pos_evol = pos_logD + 1
256  pos_Xc = pos_evol + 1
257  pos_mod_n = pos_Xc + 1
258  pos_eta = pos_mod_n + 1
259 
260  print(pos_M)
261  print(corename)
262  sys.exit()
263 
264  M_ini = float(corename[pos_M][1:])
265  fov = float(corename[pos_fov][2:])
266  Z = float(corename[pos_Z][1:])
267  logD = float(corename[pos_logD][4:])
268  evol_state= corename[pos_evol]
269  Xc = float(corename[pos_Xc][2:])
270  model_number = int(corename[pos_mod_n])
271  eta = float(corename[pos_eta][3:])
272 
273  return (M_ini, fov, Z, logD, Xc, model_number, eta)

357 def get_gyre_in_search_pattern_from_hist(dir_repos, filename):
358  """
359  From the full path to the MESA history file, generate a search string for globbing GYRE input files.
360  This function replaces the "/hist/" in the input filename with "/gyre_in/", and also replaces the
361  hist suffix e.g. ".hist" with "*".
362 
363  @param dir_repos: the full path to the repository, where hist files are stored. Normally, this is
364  available from tracks.dir_repos
365  @type dir_repos: string
366  @param filename: full path to the history filename.
367  @type filename: string
368  @return: regular expression for explicitly searching for gyre_in files that are linked to a specific
369  track
370  @rtype: string
371  """
372  if '/hist/' not in filename:
373  logger.error('get_gyre_in_search_pattern_from_hist: "/hist/" not in the filename path')
374  sys.exit(1)
375 
376  if dir_repos in filename:
377  filename = filename.replace(dir_repos, '')
378  srch = filename.replace('/hist/', '/gyre_in/')
379  ind = srch.rfind('.')
380  srch = srch[:ind] + '*'
381 
382  return srch

385 def gen_histname_from_gyre_in(gyre_in_filename):
386  """
387  convert the full filename of the gyre_in file to a full path of the hist file, by following these
388  steps:
389  1. substitute "gyre_in" with "hist"
390  2. strip off the part of the filename after "logDxx.xx"
391  3. append '.hist' at the end of the file
392  """
393  f = gyre_in_filename
394  f = f.replace('gyre_in', 'hist')
395  ind_logD = f.rfind('logD')
396  ind_keep = ind_logD + 4 + 5 # 4 for logD, 5 for the value
397  f = f[:ind_keep] + '.hist'
398 
399  return f

402 def prepare_models_data(self_models):
403  """
404  Obsolete: This routine is a no Go, when dealing with the entire database, because we immediately
405  run out of memory.
406 
407  This routine prepares the necessary data needed to fill up all required fields in the "model" objects.
408  For that, we use the values from the history filenames, from GYRE input filename, and from the history
409  columns, as soon as we match the model_number of the input model with that of the evolution step in
410  the history file.
411  Note: For large number of input GYRE files, this routine is extremely inefficient, because for every
412  input model, the history file is read one time. A better approach is provided by this routine:
413  var_lib.get_hist_and_gyre_in_data().
414 
415  @param self_models: an instance of the "var_def.models" class
416  @type self_models: models object
417 
418  """
419  sm = self_models
420 
421  sm.find_list_filenames()
422  list_gyre_in = sm.get_list_filenames()
423 
424  # fetch "model" attribute names excluding default __doc__, __init__ and __module__
425  a_model = var_def.model()
426  model_attrs = dir(a_model)
427  exclude = ['__doc__', '__init__', '__module__']
428  model_attrs = [attr for attr in model_attrs if attr not in exclude]
429 
430  # Collect all models into a list of model objects
431  list_models = []
432  for i, filename in enumerate(list_gyre_in):
433  # get an instance of the model class
434  a_model = var_def.model()
435  a_model.set_filename(filename)
436 
437  # find the corresponding history file for this model
438  histname = gen_histname_from_gyre_in(filename)
439  if not os.path.exists(histname):
440  logger.error('prepare_models_data: missing the corresponding hist file {0}'.format(histname))
441  sys.exit(1)
442  hdr, hist = read.read_mesa_ascii(histname)
443 
444  tup_gyre_in_par = get_model_parameters_from_gyre_in_filename(filename)
445 
446  M_ini = tup_gyre_in_par[0]
447  fov = tup_gyre_in_par[1]
448  Z = tup_gyre_in_par[2]
449  logD = tup_gyre_in_par[3]
450  evol_state= tup_gyre_in_par[4]
451  Xc = tup_gyre_in_par[5]
452  model_number = tup_gyre_in_par[6]
453 
454  # get the corresponding row for this model from the hist recarray
455  ind_row = model_number - 1
456  if model_number == hist['model_number'][ind_row]:
457  pass
458  else:
459  ind_row = np.where(hist['model_number'] == model_number)[0]
460  row = hist[ind_row]
461 
462  # manually, insert the 6 above attributes to the model
463  setattr(a_model, 'M_ini', M_ini)
464  setattr(a_model, 'fov', fov)
465  setattr(a_model, 'Z', Z)
466  setattr(a_model, 'logD', logD)
467  setattr(a_model, 'Xc', Xc)
468  setattr(a_model, 'model_number', model_number)
469 
470  for attr in model_attrs:
471  if attr in ['M_ini', 'fov', 'Z', 'logD', 'Xc', 'model_number', 'filename', 'set_by_dic', 'track']:
472  continue
473  else:
474  setattr(a_model, attr, row[attr])
475 
476  # generate a track object, and insert it into the model
477  the_track = var_def.track(M_ini=M_ini, fov=fov, Z=Z, logD=logD)
478  # setattr(a_model, 'track', the_track)
479  a_model.set_track(the_track)
480 
481  list_models.append(a_model)
482 
483  # store the list of model objects into the instance of the "models" class
484  sm.set_list_models(list_models)
485 
486  logger.info('Done')
