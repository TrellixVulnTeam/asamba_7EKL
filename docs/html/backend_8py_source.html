65  """
66  Make a trial attempt to the connection port, passed as "dbname", and assert if the
67  connection is possible (returns True) or not (returns False). If successful, we set
68  the connection name in the backend instance of the sampling() class.
69 
70  @param dbname: The full name of the connection port, e.g. 'grid' for local machine.
71  This value is passed by the frontend.GUI.dbname attribute
72  @type dbname: str
73  @return: True if the connection is possible, and False, otherwise
74  @rtype: bool
75  """
76  if not isinstance(dbname, str):
77  logger.error('do_connect: The input argument must be a string')
78  sys.exit(1)
79 
80  if db_def.exists(dbname):
81  # bk_sample.set('dbname', dbname)
82  BackEndSession.set('dbname', dbname)
83  return True
84  else:
85  return False
