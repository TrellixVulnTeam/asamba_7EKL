21 def hist_learning_x(self, figure_name):
22  """
23  The learning set goes through a (strict) filtering, adapted to the star (e.g. based on the observed
24  frequency/period regularities/spacing, etc.). As a result of that, only a subset of models survive
25  this. Therefore, it is very intuitive to take a look at a histogram/distribution of the input parameters
26  that survived this, and perhaps make a very rough/preliminary conclusion on the range of accepted
27  parameters that roughly reproduces the observations. In principle, this plotting routine provides
28  this functionality.
29 
30  @param self: an instance of the sampler.sampling class
31  @type self: object
32  @param figure_out: the full path to the figure to be saved on disk.
33  @type figure_out: str
34  """
35  if not self.learning_done:
36  logger.error('hist_learning_x: Call build_learning_sets() method first')
37  sys.exit(1)
38 
39  x = self.get('learning_x')
40  m, n = x.shape
41  names = self.get('feature_names')
42 
43  fig, tup_ax = plt.subplots(2, 3, figsize=(9, 6))
44  plt.subplots_adjust(left=0.05, right=0.98, bottom=0.10, top=0.97, wspace=0.2, hspace=0.25)
45  tup_ax = tup_ax.reshape(-1)
46 
47  for i, name in enumerate(names):
48  ax = tup_ax[i]
49  feature = x[:, i]
50  mn, mx = np.min(feature), np.max(feature)
51  ax.hist(feature, bins=old_div(m,10), histtype='stepfilled')
52 
53  # cosmetics
54  ax.set_xlim(mn*0.95, mx*1.02)
55  ax.set_xlabel(r'{0}'.format(name))
56 
57  plt.savefig(figure_name)
58  print(' - plot_sampler: hist_learning_x: saved {0}'.format(figure_name))
59  plt.close()
