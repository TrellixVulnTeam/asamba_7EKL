175 def write_tracks_parameters_to_ascii(self_tracks, ascii_out):
176  """
177  Store the four parameters of the MESA tracks (mass, overshoot, metallicity and extra mixing) as
178  an ascii file on the disk. To do so, the var_def.get_track_parameters() method must have already
179  been applied on the var_def.tracks() class object.
180  The format of the stored file is the following: the parameters in each row correspond to one track.
181  There will be four columns, separated by a single space, and they correspond to the initial mass
182  (M_ini), core overshooting parameter (fov), metallicity (Z), and extra diffusive mixing (logD),
183  respectively.
184 
185  @param self_tracks: an instance of the var_def.tracks()
186  @type self_tracks: class object
187  @param ascii_out: full path to store the track parameters.
188  @type ascii_out: string
189  """
190  if self_tracks.n_tracks == 0:
191  logger.error('write_tracks_parameters_to_ascii: No track data stored. Call get_track_parameters() first')
192  sys.exit(1)
193 
194  # add a header
195  lines = ['{0:<6s} {1:<5s} {2:<5s} {3:<5s} \n'.format('M_ini', 'fov', 'Z', 'logD')]
196 
197  list_tracks = self_tracks.list_tracks
198  for i, obj in enumerate(list_tracks):
199  str_M_ini = '{0:06.3f}'.format(obj.M_ini)
200  str_fov = '{0:05.3f}'.format(obj.fov)
201  str_Z = '{0:05.3f}'.format(obj.Z)
202  str_logD = '{0:05.2f}'.format(obj.logD)
203  line = '{0} {1} {2} {3} \n'.format(str_M_ini, str_fov, str_Z, str_logD)
204  lines.append(line)
205 
206  with open(ascii_out, 'w') as w: w.writelines(lines)
207  logger.info('write_tracks_parameters_to_ascii saved {0}'.format(ascii_out))
208  print(' - write: write_tracks_parameters_to_ascii saved {0}'.format(ascii_out))
209 
210  return True
