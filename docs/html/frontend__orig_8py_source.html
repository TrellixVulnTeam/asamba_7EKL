306  def connection_choice(self):
307  """ Receive the 3 possible connection choices in int format """
308  choice = self.connection.get()
309  self.connection.set(choice)
310  if choice == self.conn_loc: message = 'Connecting to the local machine (developer)'
311  if choice == self.conn_ivs: message = 'Connecting from inside the Institute of Astronomy (KULeuven)'
312  if choice == self.conn_https: message = 'Connecting to a remote server via internet'
313  self.update_status_bar(message)
314 
315  def check_connection(self):
316  """ Try connecting to the database based on the user's choice """
317  choice = self.connection.get()
318  if choice not in [self.conn_loc, self.conn_ivs, self.conn_https]:
319  self.update_status_bar('The connection choice is not made yet. Please choose one option.')
320  self.conn_status = False
321  else: # attempt a connection test
322  if choice == self.conn_loc: self.dbname = self.conn_loc_str
323  if choice == self.conn_ivs: self.dbname = self.conn_ivs_str
324  if choice == self.conn_https: self.dbname = self.conn_https_str
325  self.do_connect()
326  self.update_connection_state()
327

431  self.enter_log_Teff['bg'] = 'red'
432  self.update_status_bar('log(Teff)={0} is regjected. Only 3 <= log(Teff) <= 5 is accepted'.format(val) )
433  else:
434  val_OK = True
435  self.enter_log_Teff['bg'] = 'white'
436  self.obs_log_Teff.set(val)
437 
438  err_OK = False
439  if not (0 <= err <= 4):
440  self.enter_log_Teff_err['bg'] = 'red'
441  self.update_status_bar('err[log(Teff)]={0} is rejected. Only 0 <= err <= 4 is accepted'.format(err))
442  else:

445  self.obs_log_Teff_err.set(err)
