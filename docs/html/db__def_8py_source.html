126  def execute_many(self, cmnd, values, commit=True):
127  """
128  **Execute AND commit** many (a list of) SQL commands on the cursor.
129  The command is passed by the "cmnd", and the corresponding values are passed
130  by the "values" tuple. This function is very useful for inserting data into
131  the database.
132 
133  @param cmnd: A general command to execute many. E.g., cmnd can look like the
134  the following: cmnd = 'insert into table_name (var1, var2) values (?, ?)'
135  This ensures that the execute/commit process is protected against possible
136  Injection Attacks.
137  @type cmnd: string
138  @param values: A list of tuples to execute the command. For every execute/commit
139  transaction, one tuple must be in this list. The order of the quantities
140  in each tuple must match the order of the parameters in the command.
141  @type values: list of tuples
142  """
143  if not isinstance(values, list):
144  logger.error('execute_many: the 2nd argument (i.e. values) must be a list of tuples.')
145  sys.exit(1)
146 
147  n_vals = len(values)
148  if n_vals == 0:
149  logger.error('execute_many: the list of values is empty')
150  sys.exit(1)
151 
152  cursor = self.get_cursor()
153  try:
154  result = cursor.executemany(cmnd, values)
155  except psycopg2.Error as err:
156  print('error has occured:', err)
157  if result is not None:
158  logger.error('execute_many failed')
159  sys.exit(1)
160  if commit == True: self.commit()

208 def exists(dbname):
209  """
210  Check if the database already exists.
211  Returns True if the database exists, and False otherwise.
212  """
213  cmnd = 'psql -lqt | cut -d \| -f 1 | grep -w {0}'.format(dbname)
214  exe = subprocess.Popen(cmnd, shell=True, universal_newlines=True,
215  stdout=subprocess.PIPE, stderr=subprocess.PIPE)
216  err = exe.returncode
217 
218  if err is not None:
219  logger.info('grid_db.exists(): Command failed: "{0}"'.format(cmnd))
220  return False
221  else:
222  stdout = exe.stdout.read().rstrip('\r\n').strip()
223  stderr = exe.stderr.read().rstrip('\r\n').strip()
224  try:
225  assert stdout == dbname
226  return True
227  except AssertionError:
228  return False
