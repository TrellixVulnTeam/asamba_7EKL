335  def check_connection(self):
336  """ Try connecting to the database based on the user's choice """
337  choice = self.connection.get()
338  if choice not in [self.conn_loc, self.conn_ivs, self.conn_https]:
339  self.update_status_bar('The connection choice is not made yet. Please choose one option.')
340  self.conn_status.set(False)
341  else: # attempt a connection test
342  if choice == self.conn_loc: self.dbname = self.conn_loc_str
343  if choice == self.conn_ivs: self.dbname = self.conn_ivs_str
344  if choice == self.conn_https: self.dbname = self.conn_https_str
345  self.do_connect()
346  self.update_connection_state()

397  def plot_observed_frequencies(self):
398  """ If reading in put frequencies is successfull, the modes will be plotted, as a reward! """
399  from asamba.star import d_to_sec
400  if self.figure_is_shown.get(): self.figure_object.clf()
401 
402  fig, (top, bot) = plt.subplots(2, 1, figsize=(5, 5), dpi=100, tight_layout=True)
403 
404  modes = BackEndSession.get('modes')
405  freqs = np.array([ mode.freq for mode in modes ])
406  freq_errs = np.array([ mode.freq_err for mode in modes ])
407  amp = np.array([ mode.amplitude for mode in modes ])
408  log_amp = np.log10(amp)
409  per = old_div(1.,freqs)
410  per_err = old_div(freq_errs, freqs**2)
411  arr_l = np.array([ mode.l for mode in modes ])
412  arr_m = np.array([ mode.m for mode in modes ])
413  arr_l_m = list(zip(arr_l, arr_m))
414  set_l_m = set(arr_l_m)
415  n_l_m = len(set_l_m)
416  arr_p_mode= np.array([ mode.p_mode for mode in modes ])
417  arr_g_mode= np.array([ mode.g_mode for mode in modes ])
418  arr_in_df = np.array([ mode.in_df for mode in modes ])
419  arr_in_dP = np.array([ mode.in_dP for mode in modes ])
420 
421  colors = cycle(['k', 'b', 'r', 'g', 'y', 'c', 'p']) # one color for one (l, m)
422  use_log_amp = np.max(log_amp) - np.min(log_amp) > 1
423  _amp = log_amp if use_log_amp else amp
424 
425  for j, tup_l_m in enumerate(set_l_m):
426  _l, _m = tup_l_m[0], tup_l_m[1]
427  i_l_m = np.where((arr_l==_l) & (arr_m==_m))[0]
428  freq_l_m= freqs[i_l_m]
429  amp_l_m = _amp[i_l_m]
430  clr = next(colors)
431 
432  top.plot([], [], color=clr, label=tup_l_m)
433  for k, frq in enumerate(freq_l_m): top.plot([frq, frq], [0, amp_l_m[k]], color=clr)
434 
435  i_l_m = np.where((arr_l==_l) & (arr_m==_m) & (arr_in_dP==True))[0]
436  _per = per[i_l_m] * d_to_sec
437  arr_dP = _per[:-1] - _per[1:]
438  bot.plot(per[:-1], arr_dP, marker='o', linestyle='solid', color=clr)
439 
440  top.set_xlabel('Frequency (per day)')
441  if use_log_amp:
442  top.set_ylabel(r'$\log_{10}$(Observed Amplitude)')
443  else:
444  top.set_ylabel('Observed Amplitude')
445  top.set_ylim(-0.02, max(_amp)*1.05)
446  leg = top.legend(loc=1, fontsize=10, frameon=False)
447 
448  bot.set_xlabel('Period (day)')
449  bot.set_ylabel('Period Spacing (sec)')
450 
451  canvas = FigureCanvasTkAgg(fig, master=self.frame_plot_modes)
452  canvas.get_tk_widget().pack(side='top', fill='both', expand=True)
453  canvas._tkcanvas.pack(side='top', fill='both', expand=True)
454  canvas.show()
455 
456  self.figure_object = fig
457  self.figure_is_shown.set(True)
