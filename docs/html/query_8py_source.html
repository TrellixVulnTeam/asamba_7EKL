31 def without_constraint(dbname, table, returned_columns=[]):
32  """
33  Prepare a query to retrieve specific columns (from "returned_columns" argument). This is a generic
34  routine that can be used on any table to retrive all rows from the table. A subset of columns can
35  be selected for retrieval through the "returned_columns" argument.
36 
37  In fact, all this routine does is similar to the following generic SQL query:
38 
39  "SELECT * FROM table"
40 
41  where the "*" can be optionally replaced with a user-specified list of strings giving the desired
42  column names of the table to retrieve.
43 
44  @param dbname: the name of the database to connect to
45  @type dbname: str
46  @param table: The name of the table where the query is going to be prepaired, and will be imosed on.
47  @type table: str
48  @param returned_columns: the list of column names that we require the values for in the output.
49  E.g. one can set returned_columns = ['id', 'id_track', 'model_number'] to get the model id,
50  the id of the evolutionary track that the model comes from, and the model_number of the model
51  when this snapshot was stored by MESA. We check the requested column names exist as the
52  keys of the "models" table
53  @type returned_columns: list of strings
54  @return: The query string which is ready to be used
55  @rtype: str
56  """
57  n_cols = len(returned_columns)
58 
59  if n_cols == 0:
60  str_cols = ' * '
61  else:
62  str_cols = ','.join(returned_columns)
63 
64  with db_def.grid_db(dbname=dbname) as the_db:
65  # check if the table exists in the database
66  if not the_db.has_table(table):
67  logger.error('without_constraint: Database "{0}" does not have the \
68  table "{1}"'.format(dbname, table))
69  sys.exit(1)
70 
71  # check that requested column names exist among the table attributes
72  attrs = the_db.get_table_columns(table)
73  if n_cols > 0:
74  for attr in returned_columns:
75  if attr.lower() not in attrs:
76  logger.error('without_constraint: Attribute "{0}" is not a \
77  valid "models" key'.format(attr))
78  sys.exit(1)
79 
80  str_cols = ','.join(returned_columns)
81 
82  the_query = 'select {0} from {1}'.format(str_cols, table)
83 
84  return the_query

87 def with_constraints(dbname, table, returned_columns=[], constraints_keys=[], constraints_ranges=[]):
88  """
89  Prepare a query to retrieve specific columns (from "returned_columns" argument) subject to the
90  "WHERE" constraints specified by the "constraints_keys" argument, within the ranges specified in
91  "constraints_ranges" list. Therefore, this is a general-purpose routine that can be used on any
92  database, with any table therein.
93 
94  E.g.
95  >>>my_query = with_constraints(dbname = 'grid', table = 'models',
96  returned_columns = ['id', 'Xc'],
97  constraints_keys = ['log_Teff', 'log_g'],
98  constraints_ranges = [[4,4.1], [3.4, 3.5]])
99  >>>with db_def.grid_db(dbname) as my_db:
100  >>> my_db.execute_one(my_query, None)
101  >>> result = my_db.fetch_all()
102  >>>print len(results) > 0
103 
104  Note: The order of items in the input arguments "constraints_keys" and "constraints_ranges" must
105  match, because they are merged into a single SQL query in the same order as passed.
106 
107  @param dbname: the name of the database to connect to
108  @type dbname: str
109  @param table: The name of the table where the query is going to be prepaired, and will be imosed on.
110  @type table: str
111  @param returned_columns: the list of column names that we require the values for in the output.
112  E.g. one can set returned_columns = ['id', 'id_track', 'model_number'] to get the model id,
113  the id of the evolutionary track that the model comes from, and the model_number of the model
114  when this snapshot was stored by MESA. We check the requested column names exist as the
115  keys of the "models" table
116  @type returned_columns: list of strings
117  @param constraints_keys: the list of keys for which we impose constraints during the querying.
118  E.g. constraints_keys = ['age', 'mass_conv_core']. Indeed, we check internally if the requested
119  key is valid.
120  @type constrains_keys: list of strings
121  @param constraints_ranges: The list which provides the list/tuple for the lower and higher range
122  for each of the keys in the previous argument "constraints_keys". Note that the ranges are
123  applied in the same order as the keys are passed, so that first key goes with first range,
124  second key goes with the second range, and so on.
125  E.g. for the two keys mentioned above, one may pass:
126  constraints_ranges = [(1e6, 5e6), (0.3, 0.5)]
127  @return: The query string which is ready to be used
128  @rtype: str
129  """
130  if not returned_columns:
131  logger.error('with_constraints: Provide non-empty list of return columns \
132  from the "{0}" attributes'.format(table))
133  sys.exit(1)
134  if not constraints_keys:
135  logger.error('with_constraints: Provide non-empty list of constraint keys \
136  from the "{0}" attributes'.format(table))
137  sys.exit(1)
138  if not constraints_ranges:
139  logger.error('with_constraints: Provide non-empty list of constraint ranges \
140  for each "{0}" attributes in the constraints_keys list'.format(table))
141  sys.exit(1)
142 
143  n_cols = len(returned_columns)
144  n_keys = len(constraints_keys)
145  n_ranges = len(constraints_ranges)
146 
147  if n_cols == 0:
148  logger.error('with_constraints: the "returned_columns" must be non-empty list')
149  sys.exit(1)
150 
151  if n_keys != n_ranges:
152  logger.error('with_constraints: Mismatch between the number of keys and ranges')
153  sys.exit(1)
154 
155  with db_def.grid_db(dbname=dbname) as the_db:
156  # check if the table exists in the database
157  if not the_db.has_table(table):
158  logger.error('with_constraints: Database "{0}" does not have the \
159  table "{1}"'.format(dbname, table))
160  sys.exit(1)
161 
162  # check that requested column names exist among the table attributes
163  attrs = the_db.get_table_columns(table)
164  for attr in returned_columns + constraints_keys:
165  if attr.lower() not in attrs:
166  logger.error('with_constraints: Attribute "{0}" is not a \
167  valid "models" key'.format(attr))
168  sys.exit(1)
169 
170  str_cols = ','.join(returned_columns)
171  str_where = []
172  for k, key in enumerate(constraints_keys):
173  low = constraints_ranges[k][0]
174  high = constraints_ranges[k][1]
175 
176  try:
177  assert low <= high
178  except AssertionError:
179  logger.error('with_constraints: for key: "{0}", lower > higher! Revert the range.')
180  sys.exit(1)
181 
182  str_where.append( '({0} between {1} and {2})'.format(key, low, high) )
183 
184  # put "AND" between each of the where statements
185  str_where = ' and '.join(str_where)
186 
187  the_query = 'select {0} from {1} where {2}'.format(str_cols, table, str_where)
188 
189  return the_query

197 def get_tracks_distinct_M_ini_logD():
198  """
199  In the database, the range of \\f$log D\f$ values were selected as a function on initial mass, so that
200  \f$\log D\f$ ranges between 0.0 and some \f$max(\log D)\f$ value in 5 discrete values; here,
201  \f$ max(\log D)\f$ is a linear function of initial mass, as:
202 
203  \f[ max(\log D_{\rm mix}) = \rm{offset} + \rm{slope}\, \times\, \log_{10}(M_{\rm ini}).
204  \f]
205  where slope=\f$(6.5 - 2.5)/(\log_{10}(35) - \log_{10}(1.4))\f$, and the
206  offset=\f$6.5-{\rm slope}\times\log_{10}(35)\f$, with 1.4 and 35 \f$M_{\odot}\f$ bening the lowest
207  and highest masses in the database, and \f$\log(D)=2.5\f$ and \f$\log(D)=6.5\f$ being the maximum logarithm
208  of diffusive mixing for the lowest and highest masses in the database, respectively.
209 
210  This routine, prepares a simple query to retrieve all combinations of (M_ini, logD) for all tracks.
211  """
212  return 'select distinct on (M_ini, logD) M_ini, logD from tracks group by M_ini, logD'

220 def get_models_id_from_M_ini_fov_Z_logD_Xc(M_ini_range, fov_range, Z_range, logD_range, Xc_range):
221  """
222  Retrieve the basic model/track attributes from the "tracks" and "models" tables in the database,
223  by providing the ranges in M_ini, fov, Z, logD and Xc, respectively.
224 
225  @param _range: the lower and upper range for searching the database for each parameter (both extremes
226  inclusive).
227  @type _range: 2-element list of floats
228  @return: a SQL search/query string
229  @rtype: str
230  """
231  try:
232  assert len(M_ini_range) == 2
233  assert M_ini_range[0] <= M_ini_range[1]
234 
235  assert len(fov_range) == 2
236  assert fov_range[0] <= fov_range[1]
237 
238  assert len(Z_range) == 2
239  assert Z_range[0] <= Z_range[1]
240 
241  assert len(logD_range) == 2
242  assert logD_range[0] <= logD_range[1]
243 
244  assert len(Xc_range) == 2
245  assert Xc_range[0] <= Xc_range[1]
246 
247  except AssertionError:
248  logger.error('get_models_id_from_M_ini_fov_Z_logD_Xc: Input range arguments must have length=2')
249  sys.exit(1)
250 
251  where_M_ini = '(M_ini between {0} and {1})'.format(M_ini_range[0], M_ini_range[1])
252  where_fov = '(fov between {0} and {1})'.format(fov_range[0], fov_range[1])
253  where_Z = '(Z between {0} and {1})'.format(Z_range[0], Z_range[1])

257 
258  the_query = 'select models.id, M_ini, fov, Z, logD, Xc from models, tracks \
259  where (models.id_track=tracks.id) and {0}'.format(where_)

313 def modes_from_fixed_id_model_id_rot(id_model, id_rot, id_type=[], freq_range=[]):
314  """
315  Return the query statement to fetch mode information (i.e. id_model, id_rot, n, id_type, freq)

324  e.g. radial modes (l, m) = (0, 0) correspond to id_type = [0], and radial prograde modes
325  correspond to [1]. Therefore, to inquire searching for radial AND dipole prograde modes, you
326  must set id_types = [0, 1]
327  @type id_types: list of integer
328  @param freq_range: the lower and upper frequency range (Hz) to scan for, regardless of the id_type

341  str_id_type = ''

343  for val in id_type:

367 def modes_from_fixed_id_model_id_rot_prepared_statement(statement, id_type=[], freq_range=[]):
368  """
369  Return the "prepared" query statement to fetch mode information (i.e. id_model, id_rot, n, id_type,
370  freq) from the "modes" table given a fixed id_model and id_rot. This means, we basically deduce mode info
371  from one GYRE output file which now sits in the database.
372 
373  @param statement: the name of the prepared statement, which is also used to execute the statement
374  @type statement: str
375  @param id_type: the list of mode identification types which are pre-defined in the grid.sql schema;
376  e.g. radial modes (l, m) = (0, 0) correspond to id_type = [0], and radial prograde modes

378  must set id_types = [0, 1]
379  @type id_types: list of integer
380  @param freq_range: the lower and upper frequency range (Hz) to scan for, regardless of the id_type
381  @type freq_range: list
382  @return: the prepared statement and the execute statment, as a tuple of two elements
383  The return rows from executing this statement will give the following
384  output tuple per row: (id_model, id_rot, n, id_type, freq)
385  @rtype: tuple
386  """
387  i_arg = 2 # args $1 and $2 are reserved for id_model and id_rot, respectively
388  dtypes = ['int', 'int']

390  if id_type:
391  for val in id_type:

399  i_arg += 1

401  dtypes.append('int')
402 
403  str_id_type = ','.join( list_str_type )
404 
405  q_id_types = ' and (id_type in ({0}))'.format(str_id_type)
406 
407  q_freq = ''

409  freq_lo = freq_range[0]
