148  def load_modes_from_file(self, filename, delimiter=''):
149  """
150  Load a file, and insert all meaningful columns in the file (i.e. those that have the same header name
151  as those of the "modes" class) into the "mode" attribute. It returns a list of modes, where each mode
152  corresponds to one line in the file. The columns are delimited based on the passed delimiter
153 
154  Strict formatting of the input file:
155  - The file has two headers as the first two lines:
156  + line 1: the name of each column
157  + line 2: the Python-intrinsic format of each line, e.g. int, float, boolean
158  - The header names must be identical to the attributes of the mode object
159  - If one attribute is unknown for all modes of the same star, that column would better be omitted.
160  E.g. if for a star we do not know the modes form a frequency spacing or not, we leave this column
161  off, instead of setting it off for all modes.
162  - if a value for a mode is unknown, the column must read None or none. Never leave an unknown column
163  empty.
164  - For boolean columns, "1" means True, and "0" means False.
165 
166  @param filename: the full path where the
167  @type filename: str
168  @param delimiter: the delimiting character between the columns, e.g. ',', or space, etc. Default: ''
169  @type delimiter: str
170  @return: list of modes
171  @rtype: list
172  """
173  modes = _load_modes_from_file(filename=filename, delimiter=delimiter)
174  n = len(modes)
175  self.set('modes', modes)
176  self.set('num_modes', n)

180 def _load_modes_from_file(filename, delimiter=''):
181  """
182  For detailed documentation, please refer to the method modes.load_modes_from_file()
183  """
184  if not os.path.exists(filename):
185  logger.error('_load_modes_from_file: The file "{0}" does not exist'.format(filename))
186  sys.exit(1)
187 
188  with open(filename, 'r') as r: lines = r.readlines()
189  n_lines = len(lines)
190  if n_lines <= 2:
191  logger.error('_load_modes_from_file: Input file must have two lines of header and at least one mode line')
192  sys.exit(1)
193 
194  header = lines.pop(0).rstrip('\r\n').split(delimiter)
195  header = [val.strip() for val in header]
196  n_hdr = len(header)
197  if n_hdr < 2:
198  logger.error('_load_modes_from_file: There must be at least two columns in the file')
199  sys.exit(1)
200 
201  types = lines.pop(0).rstrip('\r\n').split(delimiter)
202  types = [val.strip() for val in types]
203  n_types = len(types)
204  if n_types != n_hdr:
205  logger.error('_load_modes_from_file: The 1st and 2nd line must have identical number of columns!')
206  sys.exit(1)
207  conv = []
208  for k, t in enumerate(types):
209  t = t.lower()
210  if t == 'int':
211  conv.append(int)
212  elif t == 'float':
213  conv.append(float)
214  elif t == 'boolean' or t == 'bool':
215  conv.append(bool)
216  elif t == 'str':
217  conv.append(str)
218  else:
219  print('t is: {0}'.format(t), k)
220  logger.error('_load_modes_from_file: 2nd line can only have "int", "float", "str" or "bool".')
221  sys.exit(1)
222 
223  # iteratively load a mode and store in a list
224  loaded = []
225  for k, line in enumerate(lines):
226  line = line.rstrip('\r\n').split(delimiter)
227  a_mode = mode()
228 
229  for j, attr in enumerate(header):
230  val = line[j].strip()
231  if val.lower == 'none':
232  a_mode.setattr(attr, None)
233  else:
234  func = conv[j]
235  val = func(val)
236  try:
237  a_mode.set(attr, val)
238  except:
239  logger.error('_load_modes_from_file: Unrecognized attribute found:')
240  print(k, j, func, line[j], attr, val, hasattr(a_mode, attr))
241  sys.exit(1)
242 
243  loaded.append(a_mode)
244 
245  # check frequency monotonicity
246  freqs = np.array([this.freq for this in loaded])
247  d_freq = freqs[1:] - freqs[:-1]
248  if not all(df > 0 for df in d_freq):
249  logger.error('_load_modes_from_file: The mode frequencies must be strictly increasing')
250  sys.exit(1)
251 
252  return loaded

446 def _do_extra(self, attr, val):
447  """
448  Some assignemnt to the star() object requires taking care of other attributes decently. This routine
449  ensures the consistency of the assignments. E.g. if you set Teff, the log_Teff will be assigend here.
450  Note: for asymmetric errors in logarithmic scale, we use the taylor expnsion of the following form
451 
452  \f[
453  \log(x\pm\epsilon)\approx\log(x)\pm\frac{\epsilon}{(\ln10)x}-\frac{\epsilon^2}{(2\ln10)x^2}.
454  \f]
455  """
456  if attr == 'Teff' and self.log_Teff == 0:
457  self.log_Teff = np.log10(val)
458 
459  if attr == 'log_Teff' and self.Teff == 0:
460  self.Teff = np.power(10, val)
461  print('1')
462 
463  if attr == 'Teff_err_lower' and self.log_Teff_err_lower == 0:
464  if self.Teff == 0:
465  logger.error('_do_extra: Specify Teff first')
466  sys.exit(1)
467  x, eps, ln10 = self.Teff, val, np.log(10.)
468  self.log_Teff_err_lower = old_div(eps,(ln10 * x)) + old_div(eps**2,(2*ln10*x**2))
469 
470  if attr == 'Teff_err_upper' and self.log_Teff_err_upper == 0:
471  if self.Teff == 0:
472  logger.error('_do_extra: Specify Teff first')
473  sys.exit(1)
474  x, eps, ln10 = self.Teff, val, np.log(10.)
475  self.log_Teff_err_upper = old_div(eps,(ln10 * x)) - old_div(eps**2,(2*ln10*x**2))
476 
477  if attr == 'log_Teff_err_lower' and self.Teff_err_lower == 0:
478  if self.log_Teff == 0:
479  logger.error('_do_extra: Specify log_Teff and log_Teff_err_lower first')
480  sys.exit(1)
481  self.Teff_err_lower = np.power(10, self.log_Teff) - \
482  np.power(10, self.log_Teff - val)
483 
484  if attr == 'log_Teff_err_upper' and self.Teff_err_upper == 0:
485  if self.log_Teff == 0:
486  logger.error('_do_extra: Specify log_Teff and log_Teff_err_upper first')
487  sys.exit(1)
488  self.Teff_err_upper = np.power(10, self.log_Teff + val) - \
489  np.power(10, self.log_Teff)
