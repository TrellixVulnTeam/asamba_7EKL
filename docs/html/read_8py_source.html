57 def read_inlist(filename):
58  """
59  This function reads an ASCII file which specifies any set of options as a list of tuples (attr, val) for
60  valid entries in the file. It follows the same idea as the widely-used Fortran inlists.
61 
62  The user comments, specified using "#" are trimmed off from anywhere in the lines, so that one may comment
63  the inlist file in a line before the attr = val set or after it. E.g., the following two options are both
64  valid:
65 
66  # Here, I specify the name of my star
67  name = 'beta Cephei'
68 
69  or
70 
71  name = 'beta Cephei' # The name of my star
72 
73  if the val is 'True', it is set to boolean True, if it is 'False', it is set to boolean False, if it contains
74  '.', 'e+' or 'e-', it is interpreted as a fload, and otherwise, it is converted to integer. So, a great caution
75  has to be practiced when assigning values to the attributes in the inlist files.
76 
77  As a nice feature, the user can even toss in a list/tuple of values, e.g. var = [1, 2.3, True]. Each element inside
78  the list/tuple will be split (comma as a delimiter), and converted to the correct datatype by calling the
79  function convert_val().
80 
81  @param filename: full path to the inlist file
82  @type filename: str
83  @return: a list of (attr, val) tuples, where
84  """
85  if not os.path.exists(filename):
86  logger.error('read_inlist: The input file "{0}" not found'.format(filename))
87  sys.exit(1)
88 
89  with open(filename, 'r') as r: lines = r.readlines()
90 
91  options = []
92  for k, line in enumerate(lines):
93  line = line.strip() # rstrip('\r\n')
94  if '#' in line:
95  ind = line.find('#')
96  line = line[:ind]
97  if not line: continue # skip empty lines
98  if '=' not in line: continue
99  attr, val = line.split('=')
100  attr = attr.strip() # remove spaces
101  val = val.strip() # remove spaces
102 
103  if '"' in val: # look after string inputs
104  if val.count('"') == 2:
105  pass # string input detected
106  else:
107  logger.error('read_inlist: Ambiguous string detected in line: {0} in file: {1}'.format(line, filename))
108  sys.exit(1)
109  elif "'" in val:
110 
111  if val.count("'") == 2:
112  pass # string input detected
113  else:
114  logger.error('read_inlist: Ambiguous string detected in line: {0} in file: {1}'.format(line, filename))
115  sys.exit(1)
116  elif '[' in val and ']' in val:
117  i_l = val.index('[')
118  i_r = val.index(']')
119  vals = val[i_l+1 : i_r].split(',')
120  val = [convert_val(v) for v in vals]
121  elif '(' in val and ')' in val:
122  i_l = val.index('(')
123  i_r = val.index(')')
124  vals = val[i_l+1 : i_r].split(',')
125  val = [convert_val(v) for v in vals]
126  else:
127  val = convert_val(val)
128 
129  # print (k, attr, val)
130  options.append((attr, val))
131 
132  logger.info('read_inlist: Successfully read file: "{0}"'.format(filename))
133 
134  return options

137 def gyre_h5(filename):
138  """
139  Read the GYRE output HDF5 file in full detail, and return an instance of the var_def.modes() with
140  relevant attributes filled up. Thus, this routine reads the summary file or the eigenfunction file
141  conveniently. Example of use:
142 
143  >>>from asamba import read
144  >>>gyre_file = '/home/user/projects/gyre/beta_Cep.h5'
145  >>>mode_list = gyre_h5(gyre_file)
146  >>>freq = np.real( mode_list.freq )
147 
148  @param filename: full path to the output GYRE HDF5 file
149  @type filename: string
150  @return: an instance of the var_def.modes() class
151  @rtype: object
152  """
153  if not os.path.exists(filename):
154  logger.error('gyre_h5: "{0}" does not exist'.format(filename))
155  sys.exit(1)
156 
157  complex_dtype = np.dtype([('re', '<f8'), ('im', '<f8')])
158 
159  with h5py.File(filename, 'r') as h5:
160  with var_def.modes() as modes:
161  for attr_key, attr_val in zip(list(h5.attrs.keys()), list(h5.attrs.values())):
162  modes.set(attr_key, attr_val)
163  for column_key in list(h5.keys()):
164  if h5[column_key].dtype == complex_dtype:
165  column_val = h5[column_key][...]['re'] + 1j * h5[column_key][...]['im']
166  else:
167  column_val = h5[column_key][...]
168  modes.set(column_key, column_val)
169 
170  return modes

173 def read_mesa_ascii(filename):
174  """
175  Read a history or profile ascii output from MESA.
176  An example of using this function to read the file "input_file" is the following
177 
178  >>> input_file = '/home/user/my-files/The_Sun/LOGS/history.data'
179  >>> header, data = read.read_mesa_ascii(input_file)
180 
181  @param filename: full path to the input ascii file
182  @type filename: string
183  @return dictionary of the header of the file, and the record array for the data block.
184  @rtype: dictionary and numpy record array
185  """
186  if not os.path.isfile(filename):
187  logger.error('read_mesa_ascii: {0} does not exist'.format(filename))
188  sys.exit(1)
189 
190  with open(filename, 'r') as r: lines = r.readlines()
191  logger.info('read_mesa_ascii: {0} successfully read'.format(filename))
192 
193  skip = lines.pop(0)
194  header_names = lines.pop(0).rstrip('\r\n').split()
195  header_vals = lines.pop(0).rstrip('\r\n').split()
196  temp = np.array([header_vals], float).T
197  header = np.core.records.fromarrays(temp, names=header_names)
198  skip = lines.pop(0)
199  skip = lines.pop(0)
200 
201  col_names = lines.pop(0).rstrip('\r\n').split()
202  n_cols = len(col_names)
203 
204  int_columns = [ 'model_number', 'version_number', 'sch_stable', 'ledoux_stable',
205  'stability_type', 'num_zones', 'cz_zone', 'cz_top_zone',
206  'num_backups', 'num_retries', 'zone', 'nse_fraction' ]
207 
208  dtypes = []
209  for col in col_names:
210  if '_type' in col:
211  dtypes.append( (col, int) )
212  elif col in int_columns:
213  dtypes.append( (col, int) )
214  else:
215  dtypes.append( (col, float) )

217  data = []

219  if not line.rstrip('\r\n').split(): continue # skip empty lines
220  data.append(line.rstrip('\r\n').split())
221 
222  data = np.core.records.fromarrays(np.array(data, float).transpose(), dtype=dtypes)
223 
224  return header, data
