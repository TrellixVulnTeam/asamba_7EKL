2 """
3 This module provides interpolation between frequencies over a give range of input
4 stellar parameters. With this tool, it is no longer needed to compute too
5 highly-resolved grids around the best asteroseismic models. Instead, the resolved
6 models are prepared by the interpolation in between the grid points from the coarse model.
7 
8 The interpolation class is a derived/subclass of the sampler.sampling class, because we require
9 several of the methods defined there in this module. This is to ensure we minimize/suppress
10 the redundancy, and make the best runtime use of the parameters that are set therein.
11 """

283 def _collect_inputs_around_anchor(self):
284  """
285  Query the database for fixed points around the anchor point.
286  """
287  # Get the individual anchor parameter values
288  anc_names = self.get('anchor_param_names')
289  anc_vals = self.get('anchor_param_values')
290  for k, name in enumerate(anc_names):
291  val = anc_vals[k]
292  if name == 'M_ini':
293  anc_M_ini = val
294  elif name == 'fov':
295  anc_fov = val
296  elif name == 'Z':
297  anc_Z = val
298  elif name == 'logD':
299  anc_logD = val
300  elif name == 'Xc':
301  anc_Xc = val

352  # Find neighbors in M_ini by manipulating the indixes
353  def _get_ind(arr, target, n):
354  ind = np.argmin(np.abs(arr - target))
355  i_from = ind - n if ind - n >= 0 else 0
356  i_to = ind + n + 1 if ind + n + 1 < len(arr) else len(arr)

371  Z_range = [np.min(neighb_Z), np.max(neighb_Z)]

406  tup_Xcs = [tup[0] for tup in tup_Xcs]

411  neighb_Xc = arr_Xc[Xc_from : Xc_to]
412  Xc_range = [neighb_Xc.min(), neighb_Xc.max()]

441 def _collect_inputs_by_range(self):
442  """
443  This function queries the grid in order to provide as much interpolation nodes as possible. For that,
444  it looks at the observed frequencies, and cherry picks those models (M_ini, fov, ...) from the grid
445  where the observed frequencies are reproduced (based on e.g. period spacing, frequency spacing, etc.).
446  As a return, the "input_features", and "input_frequencies" attributes of the class will be updated.
447  """
448  # Get the query string for retrieving models.id based on the ranges
449  # There ranges are either set by the user, or are found after a call to collect_inputs_around_anchor()
450  M_ini_range = self.get('interp_range_M_ini')
451  fov_range = self.get('interp_range_fov')
452  Z_range = self.get('interp_range_Z')
453  logD_range = self.get('interp_range_logD')
454  Xc_range = self.get('interp_range_Xc')
455  eta_range = self.get('interp_range_eta')
456  eta_ids = self.get('interp_eta_ids')
457 
458  # Check if the _range lists are OK
459  try:
460  assert len(M_ini_range) == 2
461  assert len(fov_range) == 2
462  assert len(Z_range) == 2
463  assert len(logD_range) == 2
464  assert len(Xc_range) == 2
465  if not self.exclude_eta_column:
466  assert len(eta_range) == 2
467  assert len(eta_ids) > 0
468  except AssertionError:
469  logger.error('_collect_inputs_by_range: The input ranges do not have the proper length. Check them!')
470  sys.exit(1)
471 
472  q_models_id = query.get_models_id_from_M_ini_fov_Z_logD_Xc(M_ini_range=M_ini_range,
473  fov_range=fov_range, Z_range=Z_range,
474  logD_range=logD_range, Xc_range=Xc_range)

477  with db_def.grid_db(dbname=self.dbname) as the_db:
478  the_db.execute_one(q_models_id, None)
479  tup_ids = the_db.fetch_all()
480  models_ids = np.array([ tup[0] for tup in tup_ids ], dtype=int)
481  n_models = len(models_ids)
482  logger.info('_collect_inputs_by_range: Found "{0}" models\n'.format(n_models))
483 
484 
488  features_ = self.get_M_ini_fov_Z_logD_Xc_from_models_id(models_ids)
489  n_features = len(features_)
490  try:
491  assert n_features == n_models
492  except AssertionError:
493  logger.error('_collect_inputs_by_range: Inconsistent number of feature rows retrieved!')

497  if self.exclude_eta_column:
498  models_ids_= models_ids
499  dic_rot = db_lib.get_dic_look_up_rotation_rates_id(self.dbname)
500  ids_etas = list(dic_rot.values())
501  rot_ids_ = [min(ids_etas)] * n_features
502  rows_names = ['M_ini', 'fov', 'Z', 'logD', 'Xc']
503  stiched = features_[:]
504  else:
505  models_ids_= models_ids * n_etas # size: n_models * n_etas
506  rot_ids_ = ids_etas[eta_from : eta_to] * n_models # size: n_models * n_etas
507  eta_vals_ = eta_vals[eta_from : eta_to] * n_models # size: n_models * n_etas
508  rows_names = ['M_ini', 'fov', 'Z', 'logD', 'Xc', 'eta']
509  stiched = [features_[k] + eta_vals_[k] for k in range(n_models * n_etas)]
